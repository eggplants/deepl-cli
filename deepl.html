deepl

deepl-cli

  • DeepL Translator CLI using playwright-python

Note: This project works without DeepL API key. With DeepL API, use DeepLcom/deepl-python

Install

pip install deepl-cli

Usage

CLI

deepl -F en -T ja -s <<<'This tool is useful for me.'
# このツールは私にとって便利だ。
deepl -F ja -T en -s <<<'このツールは私にとって便利だ。'
# This tool is useful for me.

curl https://example.com | sed -nr '/^<body>/,/<\/body>/s/<[^>]+>//gp' | tr -d \\n > txt
deepl -f txt -F en -T ja
# 例文ドメイン このドメインは、文書の例文に使用するためのものです。事前の調整や許可を得ることなく、このドメインを文献で使用することができます。   詳細はこちら

$ deepl -h
usage: deepl [-h] (-f PATH | -s) -F FR -T TO [-t MS] [-v] [-V]

DeepL Translator CLI without API Key

options:
  -h, --help        show this help message and exit
  -f, --file PATH   source text file to translate (default: None)
  -s, --stdin       read source text from stdin (default: False)
  -F, --fr FR       input language (default: None)
  -T, --to TO       output language (default: None)
  -t, --timeout MS  timeout interval (default: 5000)
  -v, --verbose     make output verbose (default: False)
  -V, --version     show program's version number and exit

valid languages of `-F` / --fr`:
{'cs', 'fr', 'ru', 'hu', 'zh', 'da', 'nl', 'es', 'lv', 'nb', 'de', 'ko', 'it', 'pt', 'pl', 'et', 'ar', 'el', 'en', 'id', 'sv', 'ro', 'ja', 'uk', 'bg', 'sk', 'fi', 'tr', 'sl', 'lt'}

valid languages of `-T` / `--to`:
{'cs', 'fr', 'ru', 'hu', 'zh', 'da', 'nl', 'en-gb', 'es', 'lv', 'nb', 'de', 'ko', 'it', 'pt', 'zh-hans', 'pl', 'et', 'pt-br', 'ar', 'el', 'en', 'id', 'sv', 'ro', 'ja', 'uk', 'bg', 'en-us', 'sk', 'zh-hant', 'pt-pt', 'fi', 'tr', 'sl', 'lt'}

Package

from deepl import DeepLCLI

deepl = DeepLCLI("en", "ja")
deepl.translate("hello") #=> "こんにちわ"

If you use with asyncio, Use DeepLCLI.translate_async. See examples/async.py.

157    async def __translate(self, script: str) -> str:  # noqa: C901, PLR0915
158        """Throw a request."""
159        async with async_playwright() as p:
160            # Dry run
161            try:
162                browser = await self.__get_browser(p)
163            except PlaywrightError as e:
164                if "playwright install" in e.message:
165                    await asyncio.get_event_loop().run_in_executor(
166                        None,
167                        partial(install, p.chromium, with_deps=True),
168                    )
169                    await asyncio.get_event_loop().run_in_executor(
170                        None,
171                        install,
172                        p.chromium,
173                    )
174                    browser = await self.__get_browser(p)
175                else:
176                    raise
177
178            page = await browser.new_page()
179            page.set_default_timeout(self.timeout)
180
181            # skip loading page resources for improving performance
182            excluded_resources = ["image", "media", "font", "other"]
183            await page.route(
184                "**/*",
185                lambda route: (
186                    route.abort() if route.request.resource_type in excluded_resources else route.continue_()
187                ),
188            )
189
190            url = "https://www.deepl.com/en/translator"
191            if self.use_dom_submit:
192                await page.goto(url)
193            else:
194                script = quote(script, safe="")
195                await page.goto(f"{url}#{self.fr_lang}/{self.to_lang}/{script}")
196
197            # Wait for loading to complete
198            try:
199                page.get_by_role("main")
200            except PlaywrightError as e:
201                msg = f"Maybe Time limit exceeded. ({self.timeout} ms)"
202                raise DeepLCLIPageLoadError(msg) from e
203
204            if self.use_dom_submit:
205                # banner prevents clicking on language buttons, close the banner first
206                await page.click("button[data-testid=cookie-banner-lax-close-button]")
207                # we also expect the Chrome extension banner to show up
208                with contextlib.suppress(PlaywrightError):
209                    await page.wait_for_function(
210                        """
211                        () => document.querySelector('div[data-testid="chrome-extension-toast"]')
212                        """,
213                    )
214
215                # try to close the extension banner
216                with contextlib.suppress(PlaywrightError):
217                    await page.evaluate(
218                        """
219                        document.querySelector(
220                            'div[data-testid="chrome-extension-toast"]',
221                        ).querySelector('button').click()
222                        """,
223                    )
224
225                # select input / output language
226                await page.locator(
227                    "button[data-testid=translator-source-lang-btn]",
228                ).dispatch_event("click")
229                await (
230                    page.get_by_test_id("translator-source-lang-list")
231                    .get_by_test_id(
232                        f"translator-lang-option-{self.fr_lang}",
233                    )
234                    .dispatch_event("click")
235                )
236                await page.locator(
237                    "button[data-testid=translator-target-lang-btn]",
238                ).dispatch_event("click")
239                await (
240                    page.get_by_test_id("translator-target-lang-list")
241                    .get_by_test_id(
242                        f"translator-lang-option-{self.to_lang}",
243                    )
244                    .dispatch_event("click")
245                )
246                # fill in the form of translating script
247                await page.fill(
248                    "div[aria-labelledby=translation-source-heading]",
249                    script,
250                )
251
252            # Wait for translation to complete (perhaps partially)
253            try:
254                await page.wait_for_function(
255                    """
256                    () => document.querySelector(
257                    'd-textarea[aria-labelledby=translation-target-heading]')?.value?.length > 0
258                    """,
259                )
260            except PlaywrightError as e:
261                msg = f"Time limit exceeded. ({self.timeout} ms)"
262                raise DeepLCLIPageLoadError(msg) from e
263
264            # Get the number of lines in the translated text field
265            try:
266                line_count = await page.evaluate(
267                    """
268                    document.querySelector(
269                        'd-textarea[aria-labelledby=translation-target-heading]',
270                    ).children[0].children.length
271                    """,
272                )
273            except PlaywrightError as e:
274                msg = "Unable to evaluate line count of the translation"
275                raise DeepLCLIPageLoadError(msg) from e
276
277            # Since the site may not output all lines at once, we wait until each line is finished
278            # and then add it to the list of translated lines
279            translated_lines = []
280            for line_index in range(line_count):
281                try:
282                    await page.wait_for_function(
283                        f"""
284                        () => {{
285                            t = document.querySelector(
286                                'd-textarea[aria-labelledby=translation-target-heading]',
287                            )?.children[0]?.children[{line_index}]?.innerText ?? '';
288                            return t.length > 0 && !t.includes('[...]');
289                        }}
290                        """,
291                    )
292                except PlaywrightError as e:
293                    msg = f"Time limit exceeded for line {line_index}. ({self.timeout} ms)"
294                    raise DeepLCLIPageLoadError(msg) from e
295
296                try:
297                    translated_text = await page.evaluate(
298                        f"""
299                        document.querySelector(
300                            'd-textarea[aria-labelledby=translation-target-heading]'
301                        ).children[0].children[{line_index}].innerText
302                        """,
303                    )
304                    translated_lines.append(translated_text)
305                except PlaywrightError as e:
306                    msg = f"Unable get translated text for line {line_index}"
307                    raise DeepLCLIPageLoadError(msg) from e
308
309            # Get information
310            input_textbox = page.get_by_role("region", name="Source text").locator(
311                "d-textarea",
312            )
313            output_textbox = page.get_by_role(
314                "region",
315                name="Translation results",
316            ).locator("d-textarea")
317
318            self.translated_fr_lang = str(
319                await input_textbox.get_attribute("lang"),
320            ).split("-")[0]
321            self.translated_to_lang = str(
322                await output_textbox.get_attribute("lang"),
323            ).split("-")[0]
324
325            res = "".join(translated_lines)
326
327            await browser.close()
328
329            return res

331    def __sanitize_script(self, script: str) -> str:
332        """Check command line args and stdin."""
333        script = script.rstrip("\n")
334
335        if self.max_length is not None and len(script) > self.max_length:
336            msg = f"Limit of script is less than {self.max_length} chars (Now: {len(script)} chars)"
337            raise DeepLCLIError(msg)
338
339        if len(script) <= 0:
340            msg = "Script seems to be empty."
341            raise DeepLCLIError(msg)
342
343        return script.replace("/", r"\/").replace("|", r"\|")
