deepl

deepl-cli

  • DeepL Translator CLI using playwright-python

Note: This project works without DeepL API key. With DeepL API, use DeepLcom/deepl-python

Install

pip install deepl-cli

Usage

CLI

deepl -F en -T ja -s <<<'This tool is useful for me.'
# このツールは私にとって便利だ。
deepl -F ja -T en -s <<<'このツールは私にとって便利だ。'
# This tool is useful for me.

curl https://example.com | sed -nr '/^<body>/,/<\/body>/s/<[^>]+>//gp' | tr -d \\n > txt
deepl -f txt -F en -T ja
# 例文ドメイン このドメインは、文書の例文に使用するためのものです。事前の調整や許可を得ることなく、このドメインを文献で使用することができます。   詳細はこちら

$ deepl -h
usage: deepl [-h] (-f PATH | -s) -F FR -T TO [-t MS] [-v] [-V]

DeepL Translator CLI without API Key

options:
  -h, --help        show this help message and exit
  -f, --file PATH   source text file to translate (default: None)
  -s, --stdin       read source text from stdin (default: False)
  -F, --fr FR       input language (default: None)
  -T, --to TO       output language (default: None)
  -t, --timeout MS  timeout interval (default: 5000)
  -v, --verbose     make output verbose (default: False)
  -V, --version     show program's version number and exit

valid languages of `-F` / --fr`:
{'cs', 'fr', 'ru', 'hu', 'zh', 'da', 'nl', 'es', 'lv', 'nb', 'de', 'ko', 'it', 'pt', 'pl', 'et', 'ar', 'el', 'en', 'id', 'sv', 'ro', 'ja', 'uk', 'bg', 'sk', 'fi', 'tr', 'sl', 'lt'}

valid languages of `-T` / `--to`:
{'cs', 'fr', 'ru', 'hu', 'zh', 'da', 'nl', 'en-gb', 'es', 'lv', 'nb', 'de', 'ko', 'it', 'pt', 'zh-hans', 'pl', 'et', 'pt-br', 'ar', 'el', 'en', 'id', 'sv', 'ro', 'ja', 'uk', 'bg', 'en-us', 'sk', 'zh-hant', 'pt-pt', 'fi', 'tr', 'sl', 'lt'}

Package

from deepl import DeepLCLI

deepl = DeepLCLI("en", "ja")
deepl.translate("hello") #=> "こんにちわ"

If you use with asyncio, Use DeepLCLI.translate_async. See examples/async.py.

239    async def __translate(self, script: str) -> str:  # noqa: C901, PLR0915
240        """Throw a request."""
241        async with async_playwright() as p:
242            # Dry run
243            try:
244                browser = await self.__get_browser(p)
245            except PlaywrightError as e:
246                if "playwright install" in e.message:
247                    await asyncio.get_event_loop().run_in_executor(
248                        None,
249                        partial(install, p.chromium, with_deps=True),
250                    )
251                    await asyncio.get_event_loop().run_in_executor(
252                        None,
253                        install,
254                        p.chromium,
255                    )
256                    browser = await self.__get_browser(p)
257                else:
258                    raise
259
260            page = await browser.new_page()
261            page.set_default_timeout(self.timeout)
262
263            # skip loading page resources for improving performance
264            excluded_resources = ["image", "media", "font", "other"]
265            await page.route(
266                "**/*",
267                lambda route: (
268                    route.abort() if route.request.resource_type in excluded_resources else route.continue_()
269                ),
270            )
271
272            url = "https://www.deepl.com/en/translator"
273            if self.use_dom_submit:
274                await page.goto(url)
275            else:
276                script = quote(script, safe="")
277                await page.goto(f"{url}#{self.fr_lang}/{self.to_lang}/{script}")
278
279            # Wait for loading to complete
280            try:
281                page.get_by_role("main")
282            except PlaywrightError as e:
283                msg = f"Maybe Time limit exceeded. ({self.timeout} ms)"
284                raise DeepLCLIPageLoadError(msg) from e
285
286            if self.use_dom_submit:
287                with contextlib.suppress(PlaywrightError):
288                    await page.click("button[data-testid=cookie-banner-strict-accept-all]")
289                # we also expect the Chrome extension banner to show up
290                with contextlib.suppress(PlaywrightError):
291                    await page.wait_for_function(
292                        """
293                        () => document.querySelector('div[data-testid="chrome-extension-toast"]')
294                        """,
295                    )
296
297                # try to close the extension banner
298                with contextlib.suppress(PlaywrightError):
299                    await page.evaluate(
300                        """
301                        document.querySelector(
302                            'div[data-testid="chrome-extension-toast"]',
303                        ).querySelector('button').click()
304                        """,
305                    )
306
307                # select input / output language
308                await page.locator(
309                    "button[data-testid=translator-source-lang-btn]",
310                ).dispatch_event("click")
311                await (
312                    page.get_by_test_id("translator-source-lang-list")
313                    .get_by_test_id(
314                        f"translator-lang-option-{self.fr_lang}",
315                    )
316                    .first.dispatch_event("click")
317                )
318                await page.locator(
319                    "button[data-testid=translator-target-lang-btn]",
320                ).dispatch_event("click")
321                await (
322                    page.get_by_test_id("translator-target-lang-list")
323                    .get_by_test_id(
324                        f"translator-lang-option-{self.to_lang}",
325                    )
326                    .first.dispatch_event("click")
327                )
328                # fill in the form of translating script
329                await page.fill(
330                    "div[aria-labelledby=translation-source-heading]",
331                    script,
332                )
333
334            # Wait for translation to complete (perhaps partially)
335            try:
336                await page.wait_for_function(
337                    """
338                    () => document.querySelector(
339                    'd-textarea[aria-labelledby=translation-target-heading]')?.value?.length > 0
340                    """,
341                    timeout=self.timeout,
342                )
343            except PlaywrightError as e:
344                msg = f"Time limit exceeded. ({self.timeout} ms)"
345                raise DeepLCLIPageLoadError(msg) from e
346
347            # Get the number of lines in the translated text field
348            try:
349                line_count = await page.evaluate(
350                    """
351                    document.querySelector(
352                        'd-textarea[aria-labelledby=translation-target-heading]',
353                    ).children[0].children.length
354                    """,
355                )
356            except PlaywrightError as e:
357                msg = "Unable to evaluate line count of the translation"
358                raise DeepLCLIPageLoadError(msg) from e
359
360            # Since the site may not output all lines at once, we wait until each line is finished
361            # and then add it to the list of translated lines
362            translated_lines = []
363            for line_index in range(line_count):
364                try:
365                    await page.wait_for_function(
366                        f"""
367                        () => {{
368                            t = document.querySelector(
369                                'd-textarea[aria-labelledby=translation-target-heading]',
370                            )?.children[0]?.children[{line_index}]?.innerText ?? '';
371                            return t.length > 0 && !t.includes('[...]');
372                        }}
373                        """,
374                    )
375                except PlaywrightError as e:
376                    msg = f"Time limit exceeded for line {line_index}. ({self.timeout} ms)"
377                    raise DeepLCLIPageLoadError(msg) from e
378
379                try:
380                    translated_text = await page.evaluate(
381                        f"""
382                        document.querySelector(
383                            'd-textarea[aria-labelledby=translation-target-heading]'
384                        ).children[0].children[{line_index}].innerText
385                        """,
386                    )
387                    translated_lines.append(translated_text)
388                except PlaywrightError as e:
389                    msg = f"Unable get translated text for line {line_index}"
390                    raise DeepLCLIPageLoadError(msg) from e
391
392            # Get information
393            input_textbox = page.get_by_role("region", name="Source text").locator(
394                "d-textarea",
395            )
396            output_textbox = page.get_by_role(
397                "region",
398                name="Translation results",
399            ).locator("d-textarea")
400
401            self.translated_fr_lang = str(
402                await input_textbox.get_attribute("lang"),
403            ).split("-")[0]
404            self.translated_to_lang = str(
405                await output_textbox.get_attribute("lang"),
406            ).split("-")[0]
407
408            res = "\n".join(translated_lines)
409
410            await browser.close()
411
412            return res

414    def __sanitize_script(self, script: str) -> str:
415        """Check command line args and stdin."""
416        script = script.rstrip("\n")
417
418        if self.max_length is not None and len(script) > self.max_length:
419            msg = f"Limit of script is less than {self.max_length} chars (Now: {len(script)} chars)"
420            raise DeepLCLIError(msg)
421
422        if len(script) <= 0:
423            msg = "Script seems to be empty."
424            raise DeepLCLIError(msg)
425
426        return script.replace("/", r"\/").replace("|", r"\|")
